def F1(x):
    if(x == 0):
        return 0
    return 1 / (x / 10)
device = "cpu"
TrainSize = 10000
yrange = (-10, 10)

function = F1
loss = nn.L1Loss()
learningRate = 0.3

dataLoader = DataLoader(dataset, batch_size=100)
